        function sortearAfinidade() {
            const rand = Math.random() * 100;
            let categoriaEscolhada = "";

            if (rand < 50) {
                categoriaEscolhada = "Elementais";
            } else if (rand < 75) {
                categoriaEscolhada = "Sub-Elementais";
            } else if (rand < 85) {
                categoriaEscolhada = "Espirituais";
            } else if (rand < 95) {
                categoriaEscolhada = "Celestiais";
            } else {
                categoriaEscolhada = "Cósmicas";
            }

            const lista = afinidades[categoriaEscolhada];
            const itemSorteado = lista[Math.floor(Math.random() * lista.length)];

            document.getElementById("res-categoria").innerText = "Categoria: " + categoriaEscolhada;
            document.getElementById("res-nome").innerText = itemSorteado.nome;
            document.getElementById("res-bonus").innerText = itemSorteado.bonus;
            document.getElementById("res-passiva").innerText = itemSorteado.passiva;
            
            document.getElementById("resultado").style.display = "block";
        }
